Парсер фильмов:

Скрипт, который умеет парсить фильмы в Emacs: 

(defun print-li-list (movies-list)
    (while movies-list
       (setq name (caar movies-list))  
       (setq director (nth 2 ( car movies-list)))
       (setq year (nth 1 ( car movies-list)))
       (princ (format "<li>«%s» — режиссер %s, год %d;</li>\n"  name director year rate ))
       (setq movies-list (cdr movies-list))))

(defun print-txt-list (movies-list)
    (while movies-list
       (setq name (caar movies-list))  
       (setq director (nth 2 ( car movies-list)))
       (setq year (nth 1 ( car movies-list)))
       (princ (format "• «%s» — режиссер %s, год %d;\n"  name director year rate))
       (setq movies-list (cdr movies-list))))
